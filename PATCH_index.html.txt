<!-- ▼▼▼ PASTE BELOW BLOCK JUST BEFORE </body> IN index.html ▼▼▼ -->
<script>
  /**
   * Service Worker registration — index.html copy.
   * Identical to the game.html registration; registering from either
   * page with scope '/' ensures the same SW instance covers both.
   */
  if ('serviceWorker' in navigator) {
    window.addEventListener('load', () => {
      navigator.serviceWorker
        .register('/sw.js', { scope: '/' })
        .then((reg) => {
          console.log('[SW] Registered from index, scope:', reg.scope);
        })
        .catch((err) => {
          console.warn('[SW] Registration skipped:', err.message);
        });
    });
  }
</script>
