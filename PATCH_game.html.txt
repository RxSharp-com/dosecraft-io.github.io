<!--
  ── 7. Service Worker registration ──────────────────────────────────
  Place this <script> block just before </body> (not in <head>).
  It is shown here for reference — see the note at the bottom.

  GITHUB PAGES SAFETY NOTE:
  The SW registration is wrapped in a feature-detect + try/catch.
  If the SW fetch fails (e.g. 404 on first deploy before sw.js is
  committed), the game continues to function normally — the SW is
  purely additive. No gameplay logic is affected.

  Paste the block below into game.html just before </body>:
-->

<!-- ▼▼▼ PASTE BELOW BLOCK JUST BEFORE </body> IN game.html ▼▼▼ -->
<script>
  /**
   * Service Worker registration for Infusion Arcade PWA.
   *
   * Safe-registration pattern:
   *   - Feature-detect: skip silently if browser has no SW support.
   *   - Scoped to '/': covers both /game.html and /index.html.
   *   - try/catch: a missing sw.js (e.g. branch not yet deployed)
   *     logs a warning and never blocks the game from loading.
   *
   * COMPANION MODE TIMER SAFETY:
   *   The session timer (sessionStartTime, sessionActive) lives in
   *   localStorage and React component state — not in network state.
   *   Registering or updating a SW does NOT touch localStorage and
   *   cannot cause the timer to reset. The SW only caches file bytes.
   */
  if ('serviceWorker' in navigator) {
    window.addEventListener('load', () => {
      navigator.serviceWorker
        .register('/sw.js', { scope: '/' })
        .then((reg) => {
          console.log('[SW] Registered, scope:', reg.scope);
        })
        .catch((err) => {
          console.warn('[SW] Registration skipped:', err.message);
          // Game continues normally — SW is enhancement-only.
        });
    });
  }

  /**
   * Prevent accidental page scroll / pull-to-refresh during canvas gameplay.
   * This targets the canvas element specifically, leaving other scrollable
   * UI (drug-select lists, companion mode panels) unaffected.
   *
   * touch-action: none is also set in CSS above; this JS patch ensures the
   * canvas element receives it even if React re-renders replace its attributes.
   */
  document.addEventListener('DOMContentLoaded', () => {
    function patchCanvas() {
      const canvases = document.querySelectorAll('canvas');
      canvases.forEach((c) => {
        c.style.touchAction = 'none';
      });
    }

    // Run once on load…
    patchCanvas();

    // …and re-run whenever React mounts a new canvas (e.g. new round).
    const observer = new MutationObserver(patchCanvas);
    observer.observe(document.body, { childList: true, subtree: true });
  });
</script>
